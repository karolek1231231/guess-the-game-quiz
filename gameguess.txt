<script>
const quizData = [
    { question: "A multiplayer battle arena with champions who have unique abilities.", options: ["MOBA", "FPS", "RPG", "Battle Royale"], answer: "MOBA" },
    { question: "Open world survival game with crafting and zombies.", options: ["Survival", "Platformer", "Racing", "Puzzle"], answer: "Survival" },
    { question: "A story-driven RPG with turn-based combat.", options: ["RPG", "FPS", "MOBA", "Fighting"], answer: "RPG" },
    { question: "Fast-paced shooter where the last player standing wins.", options: ["Battle Royale", "FPS", "Puzzle", "Simulation"], answer: "Battle Royale" },
];

let currentQuestion = 0;
let score = 0;

function showQuestion() {
    if(currentQuestion >= quizData.length) {
        document.getElementById("question").innerHTML = "Quiz Completed!";
        document.getElementById("options").innerHTML = "";
        document.getElementById("score").innerHTML = "Your Score: " + score + " / " + quizData.length;
        return;
    }

    const q = quizData[currentQuestion];
    document.getElementById("question").innerText = q.question;

    const optionsDiv = document.getElementById("options");
    optionsDiv.innerHTML = "";
    q.options.forEach(option => {
        const btn = document.createElement("button");
        btn.innerText = option;
        btn.className = "option";
        btn.onclick = () => checkAnswer(option);
        optionsDiv.appendChild(btn);
    });
}

function checkAnswer(selected) {
    if(selected === quizData[currentQuestion].answer) score++;
    currentQuestion++;
    showQuestion();
}

showQuestion();
</script>
